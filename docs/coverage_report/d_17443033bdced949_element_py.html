7class Element: 

8 """A class representing a chemical element. 

9 

10 An element is chosen to be an instance for the class. Upon initialization, 

11 the instance will take properties of the chosen element from the 

12 corresponding named tuple ElementProperties held within the elements 

13 dictionary below this class, and assign them as read-only attributes. 

14 

15 Args: 

16 in_symbol: a string of one or two letters corresponding to the 

17 chemical symbol of the element from the periodic table. Elements from 

18 Z=1 to Z=118 can be selected. 

19 

20 Returns: 

21 a class instance containing the properties of the chosen element. 

22 """ 

23 

24 def __init__(self, in_symbol: str): 

25 """Initialize the element using its chemical symbol """ 

26 in_symbol = in_symbol.title() 

27 self.__verify_symbol(in_symbol) 

28 self._symbol = elements[in_symbol].symbol 

29 self._name = elements[self.symbol].name 

30 self._atomic_number = elements[self.symbol].atomic_number 

31 self._atomic_radius = elements[self.symbol].atomic_radius 

32 self._colour = elements[self.symbol].colour 

33 

34 @property 

35 def symbol(self) -> str: 

36 """Returns the chemical symbol of the element""" 

37 return self._symbol 

38 

39 @property 

40 def name(self) -> str: 

41 """Returns the name of the element""" 

42 return self._name.title() 

43 

44 @property 

45 def atomic_number(self) -> int: 

46 """Returns the atomic number of the element""" 

47 return self._atomic_number 

48 

49 @property 

50 def atomic_radius(self) -> float: 

51 """Returns the atomic radius of the element""" 

52 return self._atomic_radius 

53 

54 @property 

55 def colour(self) -> str: 

56 """Returns the hex colour code of the element""" 

57 return self._colour 

58 

59 # Note we do not use a property setter as symbol must be read-only 

60 def __verify_symbol(self, chemical_symbol: str): 

61 """Verifies that a given string represents a valid chemical symbol. 

62 

63 If successful, no action is taken. Otherwise, an appropriate exception 

64 is raised. 

65 """ 

66 if not isinstance(chemical_symbol, str): 

67 raise TypeError("You must supply the chemical symbol as a string") 

68 if chemical_symbol.title() not in elements: 

69 raise ValueError("Not a valid chemical element symbol")
